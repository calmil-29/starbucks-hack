function selectDepositOption(type, value, event) { if (event) event.stopPropagation(); // 衝撃波を止める // 今クリックされたのが、【第３モーダル】か【第２モーダル】かを自動判定する const isAutocharge = event.target.closest('.autocharge-menu-list') !== null; // 【看板の特定】何番目のメニューの文字 (.current-value) を書き換えるか決める let menuIndex = 1; if (type === 'amount') menuIndex = 2; // 入金額 if (type === 'condition') menuIndex = 3; // 入金条件（オートチャージのみ） if (type === 'method') { menuIndex = isAutocharge ? 4 : 3; } // 【看板を掴む】 const listClass = isAutocharge ? '.autocharge-menu-list' : '.deposit-menu-list'; const currentLabel = document.querySelector(`${listClass} .deposit-menu-item:nth-child(${menuIndex}) .current-value`); // 【安全ガード】もしHTML（看板）がまだ画面に存在しない場合は、ここで処理を中断する if (!currentLabel) { console.warn(`看板が見つかりません。listClass: ${listClass}, menuIndex: ${menuIndex}`); return; } // 1. 各項目のデータ更新と看板への文字反映 if (type === 'card') { currentSelectedCardType = value; // 'main' か 'sub' を金庫に報告 if (value === 'main') { currentLabel.textContent = `Starbucks Card ¥${starbucksCardBalance.toLocaleString()}`; } else if (value === 'sub') { currentLabel.textContent = `サブカード ¥${subCardBalance.toLocaleString()}`; } } else if (type === 'amount' || type === 'condition') { // 💡 窓のタイトル文字（数字に変換できないもの）を拾うバグを防ぐガード const numValue = Number(value); if (!isNaN(numValue)) { // 金庫の最新データを「数値」として確実に上書き更新する if (isAutocharge) { if (type === 'amount') autochargeAmount = numValue; if (type === 'condition') autochargeCondition = numValue; } // 看板を「¥2,000」などの綺麗な表記にして書き換える currentLabel.textContent = `¥${numValue.toLocaleString()}`; } // 💡 すっきり化：ここにいた「内側のelse」は不要なので削除したよ！ } else { // 支払い方法（method）などの時は、選ばれた文字をそのまま看板に書き換える currentLabel.textContent = value; } // 2. 【第３モーダル専用】最上部のサマリーテキストをリアルタイム更新する if (isAutocharge) { const summaryEl = document.getElementById('autoSummaryText'); if (summaryEl) { summaryEl.textContent = `¥${autochargeAmount.toLocaleString()}（入金条件：残高が¥${autochargeCondition.toLocaleString()}未満）`; } // 3. 【バリデーション】入金額が入金条件を下回っているかチェックする const errorBox = document.getElementById('autoErrorMessage'); const conditionMenu = document.querySelector('.autocharge-menu-list .deposit-menu-item:nth-child(3)'); // 入金条件の枠 const nextBtn = document.querySelector('.autocharge-next-btn'); // 次へボタン // 💡 厳格チェック：確実に数値同士で比較させることで「同じ金額なのにエラーが出るバグ」を完全回避！ if (Number(autochargeAmount) < Number(autochargeCondition)) { // 条件不一致：エラーを表示して、枠を赤くし、次へボタンを押せなくする if (errorBox) errorBox.style.display = 'block'; if (conditionMenu) conditionMenu.classList.add('has-error'); if (nextBtn) nextBtn.disabled = true; } else { // 条件クリア：エラーを消して通常状態に戻す if (errorBox) errorBox.style.display = 'none'; if (conditionMenu) conditionMenu.classList.remove('has-error'); if (nextBtn) nextBtn.disabled = false; } } // 4. 自動クローズ const upperType = type.charAt(0).toUpperCase() + type.slice(1); const windowId = isAutocharge ? `autocharge${upperType}Window` : `deposit${upperType}Window`; const targetWindow = document.getElementById(windowId); if (targetWindow) { targetWindow.classList.remove('is-open', 'is-active'); } }
