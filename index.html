// 引数の「type」の中に、'card' や 'amount' や 'method' という文字が自動で入ってきます function toggleDepositWindow(type, event) { if (event) event.stopPropagation(); // 衝撃波を止める // 1. まずは、今開いているかもしれない「他の小窓」をすべて一旦きれいに閉じます // 💡 ここで「共通のクラス名」を使うことで、全ての小窓を一斉に取得してリセットできます！ const allWindows = document.querySelectorAll('.deposit-selector-window'); allWindows.forEach(win => win.classList.remove('is-active')); // 2. 【ここが魔法！】押されたタイプに合わせて、狙い撃ちするIDの文字を合体させて作ります // 例：typeが 'amount' なら ➔ 'deposit' + 'Amount' + 'Window' ➔ '#depositAmountWindow' になる！ // 💡 英語の頭文字を大文字にするために、ちょっとした文字変換を挟みます const upperType = type.charAt(0).toUpperCase() + type.slice(1); const targetId = `deposit${upperType}Window`; // 3. 合体して作ったIDの小窓だけをピンポイントで掴んで、開閉（トグル）させます const targetWindow = document.getElementById(targetId); if (targetWindow) { targetWindow.classList.add('is-active'); } } // 引数の「type」には 'amount' など、「value」には '¥3,000' などの文字が自動で入ります function selectDepositOption(type, value, event) { if (event) event.stopPropagation(); // 衝撃波を止める // 1. 【手品の核心】タップされたメニュー（看板）の、文字を表示している場所を狙い撃ちで探します // 💡 親要素（.deposit-menu-item）をたどって、その中の「.current-value」という看板クラスを掴みます // typeが 'amount' なら、2番目のメニューアイテムの中を探しに行きます let menuIndex = 1; if (type === 'amount') menuIndex = 2; if (type === 'method') menuIndex = 3; const currentLabel = document.querySelector(`.deposit-menu-list .deposit-menu-item:nth-child(${menuIndex}) .current-value`); // 2. 看板の文字を、別窓で選ばれた新しい文字（value）にガサッと上書きします if (currentLabel) { currentLabel.textContent = value; } // 3. 文字を書き換えたら、用済みになった別窓をフワッと閉じます // 💡 前回の「IDを自動合成する魔法」をここでも使い回します const upperType = type.charAt(0).toUpperCase() + type.slice(1); const targetWindow = document.getElementById(`deposit${upperType}Window`); if (targetWindow) { targetWindow.classList.remove('is-active'); // クラスを外して閉じる } }
