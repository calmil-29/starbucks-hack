//カルーセルの要素を取得(カード裏) const carousel = document.querySelector('.ticket-carusel-inner'); const dots = document.querySelectorAll('.carousel-dots .dot'); //スクロールを監視するイベント carousel.addEventListener('scroll', () => { const scrollLeft = carousel.scrollLeft; const itemWidth = carousel.offsetWidth; const currentIndex = Math.round(scrollLeft / itemWidth); dots.forEach((dot, index) => { if(index === currentIndex) { dot.classList.add('active'); } else { dot.classList.remove('active'); } }); });
