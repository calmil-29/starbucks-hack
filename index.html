// モダールを開く関数 function openQrModal(event) { // 非常に重要な一行 : 親要素(カード全体)にクリックイベントを伝えない if(event) event.stopPropagation(); // モダールが閉じると同時に、コード番号も閉じた状態に戻す const codeNumber = document.querySelector('.code-number'); const toggleBtn = document.querySelector('.code-toggle'); // ここ追加 const selectorWindow = document.querySelector('.card-selector-window'); if (codeNumber && toggleBtn) { codeNumber.classList.remove('is-active'); // 開閉クラスを強制的に外す toggleBtn.textContent = 'コード番号を表示 v'; // 文字も初期状態に戻す } // 1.タップされた要素の「目印(payment, reward, gift のどれか)」を読み取る」 const clickedArea = event.currentTarget; const dataType = clickedArea.getAttribute('data-type'); // 2.タップされた場所のQRコードをモダール上部に拡大表示する const clickedImg = clickedArea.querySelector('img'); const qrSrc = clickedImg.getAttribute('src'); document.getElementById('modalQrImg').setAttribute('src', qrSrc); const cardZone = document.querySelector('.modal-card-zone'); const cardImg = document.getElementById('modalCardImg') // 3.条件分岐の切り替え。data-type(カードの種類)に合わせて、モダールの見た目を3パターンに切り替える if (dataType === 'payment') { // 【パターン1 : 表面の支払いカード】 cardZone.style.display = 'flex' // 入金ゾーンを出す if (cardImg) cardImg.setAttribute('src', '../images/starbucks-card-design.png'); } else if (dataType === 'rewards') { // 【パターン2 : 裏面のリワードカード】 cardZone.style.display ='none' // 入金ゾーンを隠す if (cardImg) cardImg.setAttribute('src', '../images/starbucks-rewards500-design.png') } else if (dataType === 'eGift700') { // 【パターン3 : 裏面のeGiftカード】 cardZone.style.display ='none' // 入金ゾーンを隠す if (cardImg) cardImg.setAttribute('src', '../images/starbucks-egift700-design.png') } else if (dataType === 'eGift300') { // 【パターン4 : 裏面のeGiftカード】 cardZone.style.display ='none' // 入金ゾーンを隠す if (cardImg) cardImg.setAttribute('src', '../images/starbucks-egift300-design.png') } // モダールを下から表示 document.getElementById('qrModal').classList.add('is-open'); } // モダールを閉じる function closeQrModal(event) { // 1.もし「白いエリア(modal-content)」の内側がタップされた場合は、閉じる処理を完全にブロックする if (event && event.target.closest('.modal-content') && !event.target.classList.contains('close-btn')) { event.stopPropagation(); return; } // 2. 「背景(modal-overlay)」または「×ボタン(close-btn)」がタップされた場合のみ、ここが実行されて閉じます const modal = document.getElementById('qrModal'); if (modal) { //is-open クラスを外すことで、CSSの元の状態(画面外へ隠す)に戻す modal.classList.remove('is-open'); } } //　モダール内の「コード番号を表示 v」の開閉(トグル)処理 window.addEventListener('load', () => { const toggleBtn = document.querySelector('.code-toggle'); const codeNumber = document.querySelector('.code-number'); if (toggleBtn && codeNumber) { toggleBtn.addEventListener('click', (event) => { // モダール自体が閉じてしまうのを防ぐ event.stopPropagation(); // クラスを付け外ししてアニメーションさせる codeNumber.classList.toggle('is-active'); // 文字の後ろの矢印(v と　＾)を切り分ける if (codeNumber.classList.contains('is-active')) { toggleBtn.textContent = 'コード番号を非表示 ^'; } else { toggleBtn.textContent = 'コード番号を表示 v'; } }); } }); // カードの「v」を押した時に、カード切り替え窓を開閉する関数 function toggleCardSelector(event) { // 1.安全装置 (土台のファイルがあるか、?.でスマートに確認) if(!event) return; // 2.超重要！タップの衝撃波が、上のモダール全体に伝わるのをストップする event.stopPropagation(); // 3.HTMLから「カード切り替え別窓」の箱をガシッと掴む const selectorWindow = document.querySelector('.card-selector-window'); if(selectorWindow) { // 4.「is-active」クラスがついていれば外し、なければつける(トグル) selectorWindow.classList.toggle('is-active'); } }
