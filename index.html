/ 【第２モーダル】別窓の中の選択肢をタップしたときに、文字を表の「看板」にコピーして閉じる関数 function selectDepositOption(type, value, event) { if (event) event.stopPropagation(); // 衝撃波を止める // 💡 【看板の特定】何番目のメニューの文字 (.current-value) を書き換えるか決める let menuIndex = 1; if (type === 'amount') menuIndex = 2; // 入金額なら2番目のメニュー if (type === 'method') menuIndex = 3; // 支払い方法なら3番目のメニュー // 対象のメニューアイテムの中にある、文字の入れ物（看板）をガシッっと掴む //⭐️.deposit-menu-item:nth-child(${menuIndex})てどこ？ const currentLabel = document.querySelector(`.deposit-menu-list .deposit-menu-item:nth-child(${menuIndex}) .current-value`); // 💡【手品の発動】看板のテキストを、選ばれた新しい文字（value）にガサッと上書き if (currentLabel) { currentLabel.textContent = value; } // 💡【自動クローズ】文字をコピーし終わったら、用済みになった別窓を閉じる const upperType = type.charAt(0).toUpperCase() + type.slice(1); const targetWindow = document.getElementById(`deposit${upperType}Window`); if (targetWindow) { targetWindow.classList.remove('is-active'); // クラスを外す } }
