// 【第2モーダル】最終確認画面の「入金する」ボタンを押したときの関数 function executeDeposit(event) { if (event) event.stopPropagation(); // 1.入力画面から「選ばれた金額」を文字として取得（例: "3,000"） const selectedAmountText = document.querySelector('.deposit-menu-list .deposit-menu-item:nth-child(2) .current-value')?.textContent || '0'; console.log("「取得した文字」はこれです →", selectedAmountText); // 2.メイン表示の「現在の残高」を表示している要素をガシッと掴む const mainBalanceEl = document.getElementById('mainCardBalance'); if (!mainBalanceEl) { alert("メイン画像の残高要素 (id='mainBalance) が見つかりません"); return; } // 3.カンマ（,）を取り除き、文字から「純粋な数値」に変換する const depositAmount = parseInt(selectedAmountText.replace(/,/g, ''), 10); const currentBalance = parseInt(mainBalanceEl.textContent.replace(/,/g,''), 10); // 4.計算（現在の残高 + 入金額） const newBalance = currentBalance + depositAmount; // 5.計算結果をカンマ付きの文字に戻して、メイン画像を書き換える mainBalanceEl.textContent = newBalance.toLocaleString(); // 6.入金が完了したので、第２モーダル全体を閉じる（is-openを剥がす） const depositModal = document.getElementById('depositModal'); if (depositModal) { depositModal.classList.remove('is-open'); } // 完了メッセージを表示 alert(`¥${depositAmount.toLocaleString()} の入金が完了しました!`); }
