const dataType = clickedArea.getAttribute('data-type'); // 2.タップされた場所のQRコードをモダール上部に拡大表示する const clickedImg = clickedArea.querySelector('img'); const qrSrc = clickedImg.getAttribute('src'); document.getElementById('modalQrImg').setAttribute('src', qrSrc); } else if (dataType === 'rewards') { // 【パターン2 : 裏面のリワードカード】 cardZone.style.display ='none' // 入金ゾーンを隠す if (cardImg) cardImg.setAttribute('src', '../images/starbucks-rewards500-design.png') } else if (dataType === 'eGift700') { // 【パターン3 : 裏面のeGiftカード】 cardZone.style.display ='none' // 入金ゾーンを隠す if (cardImg) cardImg.setAttribute('src', '../images/starbucks-egift700-design.png') } else if (dataType === 'eGift300') { // 【パターン4 : 裏面のeGiftカード】 cardZone.style.display ='none' // 入金ゾーンを隠す if (cardImg) cardImg.setAttribute('src', '../images/starbucks-egift300-design.png') } // モダールを下から表示 document.getElementById('qrModal').classList.add('is-open'); return; } // 2. 「背景(modal-overlay)」または「×ボタン(close-btn)」がタップされた場合のみ、ここが実行されて閉じます const modal = document.getElementById('qrModal'); if (modal) { //is-open クラスを外すことで、CSSの元の状態(画面外へ隠す)に戻す modal.classList.remove('is-open');
