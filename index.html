// ==================================================== // 別枠のカードをタップした時に、見た目と金額を切り替える関数 // ==================================================== function switchCard(event) { // 1.安全装置(土台のファイルがあるか確認) if (!event) return; // 2.超重要！モダール自体や別枠が閉じてしまうのを防ぐための防波堤 event.stopPropagation(); // 3.タップされた「選択肢(箱全体)」をガシッと掴む const clickedItem = event.currentTarget; // 4.タップされたカードの「目印(main-card か sub-card か)」を読み取る // HTMLの「data-change-item」に合わせて指定 const changeType = clickedItem.getAttribute('data-change-type') || clickedItem.getAttribute('data-change-item'); //5. 掴む要素(モダールの中身 + 表のメイン画像の中身) const modalCardImg = document.getElementById('qrModalCardImg') const modalBalance = document.querySelector('.qr-card-balance') const mainCardImg = document.getElementById('mainCardImg') // 表の画像 const mainCardBalance = document.getElementById('mainCardBalance') //　表の金額 // 6.条件分岐：目印に合わせて、画面と金額を書き換える if (changeType === 'main-card') { // --- メインカード(通常)への切り替え --- const mainSrc = '../images/my-card-main.svg'; const mainTxt ='00,000' if (modalCardImg) modalCardImg.setAttribute('src', mainSrc); if (modalBalance) modalBalance.textContent = '¥' + mainTxt; if (mainCardImg) mainCardImg.setAttribute('src', mainSrc); //　表も切り替え if (mainCardBalance) mainCardBalance.textContent = mainTxt; // 表も切り替え } else if (changeType === 'sub-card') { // サブカードへの切り替え const subSrc = '../images/my-card-sub.png'; const subTxt = '3,500'; if (modalCardImg) modalCardImg.setAttribute('src', subSrc); if (modalBalance) modalBalance.textContent = '¥' + subTxt; if (mainCardImg) mainCardImg.setAttribute('src', subSrc); if (mainCardBalance) mainCardBalance.textContent = subTxt; } // ７.切り替えが終わったら、カード選択の別枠を自動でシュッと閉じる const selectorWindow = document.querySelector('.qr-card-selector-window'); selectorWindow?.classList.remove('is-active') }
