function goToDepositConfirm(event) { if (event) event.stopPropagation(); // 1番目のメニュー(カード選択)、２番目のメニュー(入金額)、３番目のメニュー(支払い方法)の文字を取得 const currentCard = document.querySelector('.deposit-menu-list .deposit-menu-item:nth-child(1) .current-value')?.textContent || 'Starbucks Card'; const currentAmount = document.querySelector('deposit-menu-list .deposit-menu-item:nth-child(2) .current-value')?.textContent || '1,000'; const currentMethod = document.querySelector('deposit-menu-list .deposit-menu-item:nth-child(3) .current-value')?.textContent || '未登録'; // HTML側にある、確認画面の各IDの入れ物をガシッと掴む const confirmCard = document.getElementById('confirmvCard'); const confirmMethod = document.getElementById('confirmMethot'); const finalBtn = document.getElementById('finalDepositBtn'); // 掴んだ入れ物の中に、さっき盗んだ文字をコピーして貼り付ける if (confirmCard) { confirmCard.textContent = currentCard; } if (confirmMethod) { confirmMethod.textContent = currentMethod; } if (finalBtn) { // バッククォートを使って、選ばれた金額をボタンの文字に合体させる(例：「¥3,000 入金する」) finalBtn.textContent = `${currentAmount} 入金する`; } // 【画面のスライド発動】 // 確認画面の箱（.step-confirm）を掴んで、is-active　クラスを付与 const confirmBox = document.querySelector('.step-confirm'); if (confirmBox) { confirmBox.classList.add('is-active'); } }
