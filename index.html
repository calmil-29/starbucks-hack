// モダールを開く関数 function openQrModal(event) { // 非常に重要な一行 : 親要素(カード全体)にクリックイベントを伝えない if(event) event.stopPropagation(); // 1.タップされた要素の「目印(payment, reward, gift のどれか)」を読み取る」 const clickedArea = event.currentTarget; const dataType = clickedArea.getAttribute('data-type'); // 2.タップされた場所のQRコードをモダール上部に拡大表示する const clickedImg = clickedArea.querySelector('img'); const qrSrc = clickedImg.getAttribute('src'); document.getElementById('modalQrImg').setAttribute('src', qrSrc); const cardZone = document.querySelector('.modal-card-zone'); const cardImg = document.getElementById('modalCardImg') // 3.条件分岐の切り替え。data-type(カードの種類)に合わせて、モダールの見た目を3パターンに切り替える if (dataType === 'payment') { // 【パターン1 : 表面の支払いカード】 cardZone.style.display = 'flex' // 入金ゾーンを出す if (cardImg) cardImg.setAttribute('src', '../images/starbucks-card-design.png'); } else if (dataType === 'rewards') { // 【パターン2 : 裏面のリワードカード】 cardZone.style.display ='none' // 入金ゾーンを隠す if (cardImg) cardImg.setAttribute('src', '../images/starbucks-rewards500-design.png') } else if (dataType === 'rewards') { // 【パターン3 : 裏面のeGiftカード】 cardZone.style.display ='none' // 入金ゾーンを隠す if (cardImg) cardImg.setAttribute('src', '../images/starbucks-egift700-design.png') } else if (dataType === 'rewards') { // 【パターン4 : 裏面のeGiftカード】 cardZone.style.display ='none' // 入金ゾーンを隠す if (cardImg) cardImg.setAttribute('src', '../images/starbucks-egift300-design.png') } // モダールを下から表示 document.getElementById('qrModal').classList.add('is-open'); }
